冒泡算法：


```js
function bubbleSort(arr){
    for(let i = 0; i < arr.length - 1; i ++){
        for(let j = 0; j < arr.length - 2; j++){
            if(arr[j] > arr[j+1]){
                let temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
    return arr;
}
```
